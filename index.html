KELAS WARNA ==================== class Warna { final int r, g, b; // Constant Constructor const Warna(this.r, this.g, this.b); @override String toString() { return 'Warna(r: $r, g: $g, b: $b)'; } } void demoWarna() { print('\n=================================================='); print('PROGRAM 3: DEMO KELAS WARNA'); print('=================================================='); // Membuat objek const print('\n[1] Membuat objek dengan Constant Constructor:'); const warnaMerah = Warna(255, 0, 0); const warnaHijau = Warna(0, 255, 0); const warnaBiru = Warna(0, 0, 255); print(' Merah: $warnaMerah'); print(' Hijau: $warnaHijau'); print(' Biru : $warnaBiru'); // Demonstrasi canonicalization print('\n[2] Demonstrasi Canonicalization:'); const warnaMerah2 = Warna(255, 0, 0); print(' Apakah warnaMerah dan warnaMerah2 sama? ${identical(warnaMerah, warnaMerah2)}'); print(' Apakah warnaMerah dan warnaHijau sama? ${identical(warnaMerah, warnaHijau)}'); print('=================================================='); } // ==================== 4. KELAS TITIK ==================== // Extension untuk menghitung akar kuadrat extension Sqrt on double { double sqrt() => this > 0 ? this * this : 0; } class Titik { final double x, y; // Initializer list untuk memastikan koordinat bernilai absolut Titik(double x, double y) : x = x.abs(), y = y.abs() { print(' ✅ Titik (${this.x}, ${this.y}) berhasil dibuat.'); } // Method untuk menghitung jarak dari titik (0,0) double jarakDariNol() { return (x * x + y * y).sqrt(); } } void demoTitik() { print('\n=================================================='); print('PROGRAM 4: DEMO KELAS TITIK'); print('=================================================='); print('\n[1] Membuat titik dengan koordinat positif:'); Titik titik1 = Titik(5.0, 7.0); print(' Titik1: (${titik1.x}, ${titik1.y})'); print(' Jarak dari (0,0): ${titik1.jarakDariNol().toStringAsFixed(2)}'); print('\n[2] Membuat titik dengan koordinat negatif:'); print(' (Nilai negatif akan diubah menjadi positif oleh initializer list)'); Titik titik2 = Titik(-3.5, -8.0); print(' Titik2: (${titik2.x}, ${titik2.y})'); print(' Jarak dari (0,0): ${titik2.jarakDariNol().toStringAsFixed(2)}'); print('\n[3] Membuat titik dengan koordinat campuran:'); Titik titik3 = Titik(-2.0, 4.5); print(' Titik3: (${titik3.x}, ${titik3.y})'); print(' Jarak dari (0,0): ${titik3.jarakDariNol().toStringAsFixed(2)}'); print('=================================================='); } // ==================== 5. KELAS USER ==================== class User { String _username; String _password; String _email; User(this._username, this._password, this._email); // Getter (password tidak disediakan getter untuk keamanan) String get username => _username; String get email => _email; // Method Login bool login(String username, String password) { print('\n 🔐 Percobaan Login: "$username"'); if (username == _username && password == _password) { print(' ✅ Login BERHASIL! Selamat datang $_username.'); return true; } else { print(' ❌ Login GAGAL! Username atau password salah.'); return false; } } // Method Ganti Password dengan validasi void gantiPassword(String passwordLama, String passwordBaru) { print('\n 🔑 Percobaan Ganti Password:'); // Validasi password lama if (passwordLama != _password) { print(' ❌ Gagal: Password lama salah!'); return; } // Validasi password baru if (passwordBaru.length < 6) { print(' ❌ Gagal: Password baru minimal 6 karakter! (Panjang: ${passwordBaru.length})'); return; } if (passwordBaru == _password) { print(' ❌ Gagal: Password baru harus berbeda dari password lama!'); return; } // Jika semua validasi lolos _password = passwordBaru; print(' ✅ Password BERHASIL diubah!'); } // Method untuk menampilkan informasi user (tanpa password) void infoUser() { print(' 👤 Username: $_username'); print(' 📧 Email : $_email'); print(' 🔒 Password: [PROTECTED]'); } } void demoUser() { print('\n=================================================='); print('PROGRAM 8: DEMO KELAS USER (LOGIN SYSTEM)'); print('=================================================='); // Membuat objek User print('\n[1] Membuat user baru:'); User user1 = User('dini_putri', 'rahasia123', 'dini@email.com'); user1.infoUser(); // Demonstrasi Login print('\n[2] DEMONSTRASI LOGIN:'); print('\n Skenario 1: Login dengan data benar'); user1.login('dini_putri', 'rahasia123'); print('\n Skenario 2: Login dengan username salah'); user1.login('dini_salah', 'rahasia123'); print('\n Skenario 3: Login dengan password salah'); user1.login('dini_putri', 'password123'); // Demonstrasi Ganti Password print('\n\n[3] DEMONSTRASI GANTI PASSWORD:'); print('\n Skenario 1: Ganti password dengan data benar'); user1.gantiPassword('rahasia123', 'rahasia456'); // Berhasil print('\n Skenario 2: Login dengan password baru'); user1.login('dini_putri', 'rahasia456'); print('=================================================='); } // ==================== 9. KELAS REKENING BANK ==================== class RekeningBank { String _nomorRekening; double _saldo; RekeningBank(this._nomorRekening, this._saldo); // Getter String get nomorRekening => _nomorRekening; double get saldo => _saldo; // Method Deposit (menyetor uang) void deposit(double jumlah) { print('\n 💰 Transaksi DEPOSIT: Rp ${jumlah.toStringAsFixed(0)}'); if (jumlah > 0) { _saldo += jumlah; print(' ✅ Deposit BERHASIL!'); print(' 💳 Saldo sekarang : Rp ${_saldo.toStringAsFixed(0)}'); } else { print(' ❌ Deposit GAGAL! Jumlah harus positif (input: $jumlah)'); } } // Method Tarik Tunai (menarik uang) void tarik(double jumlah) { print('\n 💸 Transaksi TARIK TUNAI: Rp ${jumlah.toStringAsFixed(0)}'); if (jumlah <= 0) { print(' ❌ Tarik tunai GAGAL! Jumlah harus positif (input: $jumlah)'); return; } if (_saldo >= jumlah) { _saldo -= jumlah; print(' ✅ Tarik tunai BERHASIL!'); print(' 💳 Saldo sekarang : Rp ${_saldo.toStringAsFixed(0)}'); } else { print(' ❌ Tarik tunai GAGAL! Saldo tidak mencukupi.'); print(' 💳 Saldo tersedia : Rp ${_saldo.toStringAsFixed(0)}'); } } // Method Cek Saldo void cekSaldo() { print('\n 🏦 CEK SALDO'); print(' ========================='); print(' No. Rekening : $_nomorRekening'); print(' Saldo : Rp ${_saldo.toStringAsFixed(0)}'); print(' ========================='); } // Method Transfer ke rekening lain void transfer(RekeningBank tujuan, double jumlah) { print('\n 🔄 Transaksi TRANSFER: Rp ${jumlah.toStringAsFixed(0)}'); print(' Dari: $_nomorRekening'); print(' Ke : ${tujuan._nomorRekening}'); if (jumlah <= 0) { print(' ❌ Transfer GAGAL! Jumlah harus positif (input: $jumlah)'); return; } if (_saldo >= jumlah) { _saldo -= jumlah; tujuan._saldo += jumlah; print(' ✅ Transfer BERHASIL!'); print(' 💳 Saldo pengirim : Rp ${_saldo.toStringAsFixed(0)}'); } else { print(' ❌ Transfer GAGAL! Saldo pengirim tidak mencukupi.'); } } } void demoRekeningBank() { print('\n=================================================='); print('PROGRAM 9: DEMO KELAS REKENING BANK'); print('=================================================='); // Membuat dua rekening untuk simulasi print('\n[1] MEMBUAT REKENING BANK:'); var rekeningAndi = RekeningBank('011-00-1234567-8', 1000000); var rekeningBudi = RekeningBank('011-00-7654321-9', 500000); rekeningAndi.cekSaldo(); rekeningBudi.cekSaldo(); // Demonstrasi Deposit print('\n[2] SIMULASI DEPOSIT:'); rekeningAndi.deposit(500000); rekeningAndi.deposit(-100000); // Gagal (negatif) // Demonstrasi Tarik Tunai print('\n[3] SIMULASI TARIK TUNAI:'); rekeningAndi.tarik(300000); // Berhasil rekeningAndi.tarik(1500000); // Gagal (saldo tidak cukup) // Demonstrasi Transfer print('\n[4] SIMULASI TRANSFER:'); rekeningAndi.transfer(rekeningBudi, 200000); // Berhasil // Saldo akhir print('\n[5] SALDO AKHIR:'); rekeningAndi.cekSaldo(); rekeningBudi.cekSaldo(); print('=================================================='); } // ==================== MAIN FUNCTION ==================== void main() { print('===================================================='); print(' IMPLEMENTASI CONSTRUCTOR DAN ENKAPSULASI'); print(' NAMA: USMAN YAKIN TRIO SINADA'); print(' NIM : 251410006'); print(' KELAS: SI2A'); print('===================================================='); // Panggil semua demo demoBuku(); demoMobil(); demoWarna(); demoTitik(); demoMahasiswa(); demoPerson(); demoProduk(); demoUser(); demoRekeningBank(); print('\n===================================================='); print(' SEMUA PROGRAM TELAH DIJALANKAN'); print('===================================================='); }
