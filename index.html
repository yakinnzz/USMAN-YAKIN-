📚 1. Kelas Buku - Constructor Dasar

🚗 2. Kelas Mobil - Named Constructor

Memiliki parameterized constructor, named constructor baru() dan dariJson().

📁 mobil.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Mobil {
  String? merk;
  int? tahun;

  // Parameterized Constructor
  Mobil(this.merk, this.tahun);

  // Named Constructor: Mobil Baru
  Mobil.baru() {
    merk = 'Nissan Skyline GT-R R34';
    tahun = 2002;
  }

  // Named Constructor: dari JSON
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }
}

🎨 3. Kelas Warna - Constant Constructor

Menggunakan constant constructor untuk membuat objek immutable (tidak bisa diubah).

📁 warna.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Warna {
  final int r, g, b;

  // Constant Constructor
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return 'Warna(r: $r, g: $g, b: $b)';
  }
}

void main() {
  const warnaMerah = Warna(255, 0, 0);
  const warnaHijau = Warna(0, 255, 0);
  
  print(warnaMerah);  // Output: Warna(r: 255, g: 0, b: 0)
  print(warnaHijau);   // Output: Warna(r: 0, g: 255, b: 0)
}

📐 4. Kelas Titik - Initializer List

Menggunakan initializer list untuk memproses nilai sebelum objek dibuat.

📁 titik.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Titik {
  final double x, y;

  // Initializer list untuk memastikan koordinat bernilai absolut
  Titik(double x, double y)
      : x = x.abs(),
        y = y.abs() {
    print('Titik (${this.x}, ${this.y}) berhasil dibuat.');
  }
}

void main() {
  Titik titik1 = Titik(5.0, 7.0);   // Positif
  Titik titik2 = Titik(-3.5, -8.0); // Negatif akan jadi positif
}

👨‍🎓 5. Kelas Mahasiswa - Named Constructor

Memiliki default constructor dan named constructor dariMap() serta tanpaNama().

📁 mahasiswa.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Mahasiswa {
  String nama;
  String nim;
  String jurusan;
  String kelas;

  // Default Constructor
  Mahasiswa(this.nama, this.nim, this.jurusan, this.kelas);

  // Named Constructor: dari Map
  Mahasiswa.dariMap(Map<String, String> data)
      : nama = data['nama']!,
        nim = data['nim']!,
        jurusan = data['jurusan']!,
        kelas = data['kelas']!;

  // Named Constructor: Tanpa Nama (Anonim)
  Mahasiswa.tanpaNama(this.nim, this.jurusan, this.kelas) : nama = '*** ANONIM ***';
}

👤 6. Kelas Person - Enkapsulasi Dasar

Menerapkan enkapsulasi dengan getter dan setter untuk atribut private.

📁 person.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Person {
  String _nama; // Private field
  int _umur;    // Private field

  Person(this._nama, this._umur);

  // Getter
  String get nama => _nama;
  int get umur => _umur;

  // Setter dengan validasi
  set nama(String namaBaru) {
    if (namaBaru.isNotEmpty) {
      _nama = namaBaru;
    }
  }

  set umur(int umurBaru) {
    if (umurBaru > 0 && umurBaru < 150) {
      _umur = umurBaru;
    }
  }
}

🔐 8. Kelas User - Sistem Login

Enkapsulasi dengan metode login dan ganti password.

📁 user.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class User {
  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email);

  // Getter (password tidak disediakan)
  String get username => _username;
  String get email => _email;

  // Method Login
  bool login(String username, String password) {
    return username == _username && password == _password;
  }

  // Method Ganti Password
  void gantiPassword(String passwordLama, String passwordBaru) {
    if (passwordLama == _password && passwordBaru.length >= 6) {
      _password = passwordBaru;
    }
  }
}

🏦 9. Kelas RekeningBank - Simulasi Transaksi

Enkapsulasi dengan metode deposit, tarik tunai, dan transfer.

📁 rekening_bank.dart
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo);

  // Getter
  String get nomorRekening => _nomorRekening;
  double get saldo => _saldo;

  // Method Deposit
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
    }
  }

  // Method Tarik Tunai
  void tarik(double jumlah) {
    if (jumlah > 0 && _saldo >= jumlah) {
      _saldo -= jumlah;
    }
  }

  // Method Transfer
  void transfer(RekeningBank tujuan, double jumlah) {
    if (jumlah > 0 && _saldo >= jumlah) {
      _saldo -= jumlah;
      tujuan._saldo += jumlah;
    }
  }
}

  • Pemakaian constructor dan enkapsulasi membuat kode lebih terstruktur, modular, dan mudah dipelihara.
