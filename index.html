1️⃣ CONSTRUCTOR CHAINING LANJUTAN

Pengertian: Constructor yang memanggil constructor lain dalam satu kelas. Berguna untuk menghindari duplikasi kode inisialisasi.

Kata kunci: Menggunakan this() untuk memanggil constructor lain.

💡 Catatan: Constructor chaining memastikan semua inisialisasi dilakukan secara konsisten, apapun constructor yang dipanggil.
📁 constructor_chaining.dart DART 
class Produk {
  String nama;
  double harga;
  String kategori;
  
  // Constructor utama
  Produk(this.nama, this.harga, this.kategori) {
    print('Constructor utama dipanggil');
  }
  
  // Redirecting constructor - memanggil constructor utama
  Produk.elektronik(String nama, double harga) : this(nama, harga, 'Elektronik');
  
  // Redirecting constructor lainnya
  Produk.makanan(String nama, double harga) : this(nama, harga, 'Makanan');
  
  // Constructor dengan parameter berbeda
  Produk.diskon(String nama, double harga)
      : this(nama, harga * 0.9, 'Diskon');
  
  void info() {
    print('$nama - Rp$harga ($kategori)');
  }
}

void main() {
  var tv = Produk.elektronik('Samsung TV', 5000000);
  var mie = Produk.makanan('Indomie', 3500);
  var diskon = Produk.diskon('Smartphone', 3000000);
  
  tv.info();
  mie.info();
  diskon.info();
}

2️⃣ FACTORY CONSTRUCTOR LANJUTAN

Pengertian: Constructor yang tidak selalu membuat instance baru. Bisa mengembalikan instance yang sudah ada (caching) atau instance dari subclass.

Kata kunci: Menggunakan kata factory sebelum constructor.

💡 Contoh nyata: Factory constructor digunakan di Flutter untuk caching objek-objek berat seperti gambar atau koneksi database.
📁 factory_constructor.dart DART 
// Contoh 1: Singleton Pattern
class KoneksiDatabase {
  // Instance static private
  static final KoneksiDatabase _instance = KoneksiDatabase._internal();
  
  // Private constructor
  KoneksiDatabase._internal() {
    print('🔌 Koneksi database dibuat (hanya sekali)');
  }
  
  // Factory constructor mengembalikan instance yang sama
  factory KoneksiDatabase() {
    return _instance;
  }
  
  void query(String sql) {
    print('📊 Menjalankan: $sql');
  }
}

// Contoh 2: Caching
class Pengguna {
  final String id;
  final String nama;
  
  // Cache static
  static final Map<String, Pengguna> _cache = {};
  
  // Private constructor
  Pengguna._(this.id, this.nama);
  
  // Factory constructor dengan caching
  factory Pengguna.dariId(String id, String nama) {
    if (_cache.containsKey(id)) {
      print('✅ Mengambil dari cache: $id');
      return _cache[id]!;
    } else {
      print('🆕 Membuat baru: $id');
      final user = Pengguna._(id, nama);
      _cache[id] = user;
      return user;
    }
  }
}

void main() {
  print('--- SINGLETON ---');
  var db1 = KoneksiDatabase();
  var db2 = KoneksiDatabase();
  print('db1 == db2? ${identical(db1, db2)}');
  
  print('\n--- CACHING ---');
  var user1 = Pengguna.dariId('001', 'Alice');
  var user2 = Pengguna.dariId('001', 'Alice');
  var user3 = Pengguna.dariId('002', 'Bob');
}

3️⃣ PRIVATE CONSTRUCTOR & UTILITY CLASS LANJUTAN

Pengertian: Constructor yang tidak bisa diakses dari luar kelas. Biasanya untuk kelas yang hanya berisi method static (utility class).

Cara: Menggunakan underscore _ di depan nama constructor.

💡 Kegunaan: Utility class seperti Math di Java atau StringUtils - tidak perlu diinstansiasi.
📁 utility_class.dart DART 
class StringHelper {
  // Private constructor
  StringHelper._();
  
  // Static methods
  static String kapitalisasi(String teks) {
    if (teks.isEmpty) return teks;
    return teks[0].toUpperCase() + teks.substring(1).toLowerCase();
  }
  
  static String balik(String teks) {
    return teks.split('').reversed.join('');
  }
  
  static int hitungVokal(String teks) {
    final vokal = 'aiueoAIUEO';
    return teks.split('').where((c) => vokal.contains(c)).length;
  }
  
  static bool isPalindrom(String teks) {
    return teks == balik(teks);
  }
}

class Validator {
  Validator._();
  
  static bool isEmail(String email) {
    return email.contains('@') && email.contains('.');
  }
  
  static bool isNomerHP(String nomer) {
    final regex = RegExp(r'^[0-9]{10,12}$');
    return regex.hasMatch(nomer);
  }
  
  static bool isPasswordKuat(String password) {
    return password.length >= 8 &&
           password.contains(RegExp(r'[A-Z]')) &&
           password.contains(RegExp(r'[0-9]'));
  }
}

void main() {
  // var helper = StringHelper(); // ERROR! Tidak bisa diinstansiasi
  
  print('Kapitalisasi: ${StringHelper.kapitalisasi("jOHN")}');
  print('Balik: ${StringHelper.balik("DART")}');
  print('Vokal: ${StringHelper.hitungVokal("Indonesia")}');
  print('Palindrom: ${StringHelper.isPalindrom("kodok")}');
  
  print('\nValidasi:');
  print('Email: ${Validator.isEmail("user@email.com")}');
  print('No HP: ${Validator.isNomerHP("081234567890")}');
  print('Password: ${Validator.isPasswordKuat("Password123")}');
}

4️⃣ COMPUTED PROPERTIES (GETTER TANPA FIELD) LANJUTAN

Pengertian: Property yang nilainya dihitung saat diakses, bukan disimpan sebagai field.

Keuntungan: Hemat memori, nilai selalu update, logika terpusat.

💡 Analogi: Seperti "usia" seseorang - tidak perlu disimpan, cukup dihitung dari tanggal lahir setiap kali ditanya.
📁 computed_properties.dart DART 
class PersegiPanjang {
  double panjang;
  double lebar;
  
  PersegiPanjang(this.panjang, this.lebar);
  
  // Computed properties
  double get luas => panjang * lebar;
  double get keliling => 2 * (panjang + lebar);
  double get diagonal => (panjang*panjang + lebar*lebar).sqrt();
  
  // Setter yang mempengaruhi field lain
  set skala(double faktor) {
    panjang *= faktor;
    lebar *= faktor;
  }
  
  String get kategori {
    if (luas < 10) return 'Kecil';
    if (luas < 50) return 'Sedang';
    return 'Besar';
  }
}

class Suhu {
  double _celcius;
  
  Suhu(this._celcius);
  
  // Getter komputasi
  double get celcius => _celcius;
  double get fahrenheit => (_celcius * 9/5) + 32;
  double get reamur => _celcius * 4/5;
  double get kelvin => _celcius + 273.15;
  
  // Setter dengan komputasi
  set fahrenheit(double f) {
    _celcius = (f - 32) * 5/9;
  }
}

void main() {
  var pp = PersegiPanjang(5, 3);
  print('Luas: ${pp.luas}');        // 15
  print('Keliling: ${pp.keliling}');  // 16
  print('Diagonal: ${pp.diagonal.toStringAsFixed(2)}');
  print('Kategori: ${pp.kategori}');
  
  pp.skala = 2;
  print('Setelah skala 2x: ${pp.luas}'); // 60
  
  var suhu = Suhu(30);
  print('30°C = ${suhu.fahrenheit}°F');
  print('30°C = ${suhu.kelvin}K');
  
  suhu.fahrenheit = 100;
  print('100°F = ${suhu.celcius}°C');
}

5️⃣ LATE INITIALIZATION (LAZY LOADING) LANJUTAN

Pengertian: Inisialisasi variabel ditunda sampai pertama kali diakses. Berguna untuk data yang berat atau tidak selalu dibutuhkan.

Kata kunci: Menggunakan late sebelum tipe data.

💡 Manfaat: Menghemat memori dan waktu startup aplikasi karena data berat hanya di-load jika benar-benar diperlukan.
📁 late_initialization.dart DART 
class LaporanKeuangan {
  // Field yang diinisialisasi secara lazy
  late List<double> _dataTransaksi;
  late String _ringkasan;
  
  LaporanKeuangan() {
    print('📁 Objek laporan dibuat (data belum di-load)');
  }
  
  // Method untuk memuat data (lazy loading)
  void _loadData() {
    print('⏳ Memuat data transaksi...');
    // Simulasi loading data berat
    _dataTransaksi = [1000000, 2500000, 3000000, 1500000, 2000000];
    _ringkasan = _generateRingkasan();
    print('✅ Data selesai dimuat');
  }
  
  String _generateRingkasan() {
    final total = _dataTransaksi.reduce((a, b) => a + b);
    final rata = total / _dataTransaksi.length;
    return 'Total: Rp$total, Rata-rata: Rp${rata.toStringAsFixed(0)}';
  }
  
  // Getter yang memicu lazy loading
  List<double> get dataTransaksi {
    if (!_dataTransaksi.late) _loadData();
    return List.unmodifiable(_dataTransaksi);
  }
  
  String get ringkasan {
    if (!_ringkasan.late) _loadData();
    return _ringkasan;
  }
  
  double get totalTransaksi {
    if (!_dataTransaksi.late) _loadData();
    return _dataTransaksi.reduce((a, b) => a + b);
  }
}

class ProfilPengguna {
  String nama;
  int umur;
  
  // late final - diinisialisasi sekali di runtime
  late final DateTime waktuLogin;
  
  ProfilPengguna(this.nama, this.umur);
  
  void login() {
    waktuLogin = DateTime.now();
    print('🕐 Login dicatat: $waktuLogin');
  }
  
  String get durasiLogin {
    final durasi = DateTime.now().difference(waktuLogin);
    return '${durasi.inMinutes} menit yang lalu';
  }
}

void main() {
  print('🚀 APLIKASI DIMULAI');
  
  var laporan = LaporanKeuangan();
  print('Objek laporan dibuat, memory masih hemat');
  
  print('\nMengakses ringkasan...');
  print(laporan.ringkasan);  // Data di-load di sini
  
  print('\nMengakses total transaksi...');
  print('Total: Rp${laporan.totalTransaksi}');  // Data sudah ada
  
  print('\n--- LATE FINAL ---');
  var user = ProfilPengguna('Budi', 25);
  user.login();
  Future.delayed(Duration(seconds: 2), () {
    print('Durasi login: ${user.durasiLogin}');
  });
}

6️⃣ IMMUTABLE CLASS (CONST CONSTRUCTOR) LANJUTAN

Pengertian: Kelas yang objeknya tidak bisa diubah setelah dibuat. Semua field harus final dan menggunakan const constructor.

Kegunaan: Value objects, data transfer objects, konfigurasi.

💡 Keuntungan: Thread-safe, mudah di-test, predictable, dan bisa di-cache.

6️⃣ IMMUTABLE CLASS (CONST CONSTRUCTOR) LANJUTAN

Pengertian: Kelas yang objeknya tidak bisa diubah setelah dibuat. Semua field harus final dan menggunakan const constructor.

Kegunaan: Value objects, data transfer objects, konfigurasi.

💡 Keuntungan: Thread-safe, mudah di-test, predictable, dan bisa di-cache.
📁 immutable_class.dart DART 
class Point {
  final int x;
  final int y;
  
  // Const constructor
  const Point(this.x, this.y);
  
  // Method mengembalikan objek baru, bukan mengubah yang lama
  Point translate(int dx, int dy) {
    return Point(x + dx, y + dy);
  }
  
  @override
  bool operator ==(Object other) {
    return other is Point && other.x == x && other.y == y;
  }
  
  @override
  int get hashCode => x.hashCode ^ y.hashCode;
  
  @override
  String toString() => 'Point($x, $y)';
}

class Alamat {
  final String jalan;
  final String kota;
  final String kodePos;
  
  const Alamat(this.jalan, this.kota, this.kodePos);
  
  Alamat copyWith({String? jalan, String? kota, String? kodePos}) {
    return Alamat(
      jalan ?? this.jalan,
      kota ?? this.kota,
      kodePos ?? this.kodePos,
    );
  }
}

class PenggunaImmutable {
  final String id;
  final String nama;
  final int umur;
  final Alamat alamat;
  
  const PenggunaImmutable({
    required this.id,
    required this.nama,
    required this.umur,
    required this.alamat,
  });
  
  PenggunaImmutable ubahNama(String namaBaru) {
    return PenggunaImmutable(
      id: id,
      nama: namaBaru,

}

void main() {
  // Const objects - bisa di-cache
  const p1 = Point(3, 4);
  const p2 = Point(3, 4);
  const p3 = Point(5, 6);
  
  print('p1: $p1');
  print('p2: $p2');
  print('p1 == p2? ${p1 == p2}');  // true (nilai sama)
  print('identical(p1, p2)? ${identical(p1, p2)}');  // true (object sama)
  
  var p4 = p1.translate(2, 3);  // Membuat object baru
  print('p1 setelah translate: $p1 (tetap)');  // Tidak berubah
  print('p4 (baru): $p4');
  
  var alamat = Alamat('Jl. Sudirman', 'Jakarta', '12345');
  var user1 = PenggunaImmutable(
    id: '001',
    nama: 'Budi',
    umur: 25,
    alamat: alamat,
  );
  
  var user2 = user1.ubahNama('Budi Santoso');  // Object baru
  print('user1.nama: ${user1.nama}');  // Tetap 'Budi'
  print('user2.nama: ${user2.nama}');  // 'Budi Santoso'
}

7️⃣ CALLABLE CLASS (METHOD CALL) LANJUTAN

Pengertian: Kelas yang objeknya bisa dipanggil seperti fungsi dengan mengimplementasikan method call().

Kegunaan: Membuat objek yang berperilaku seperti fungsi, closure, atau function pointer.

💡 Unik di Dart: Fitur ini jarang ada di bahasa pemrograman lain! Objek bisa dipanggil langsung seperti fungsi.
📁 callable_class.dart DART 
class Greeter {
  final String salam;
  
  Greeter(this.salam);
  
  // Method call - membuat objek bisa dipanggil
  String call(String nama) {
    return '$salam, $nama!';
  }
}

class Validator {
  final int minPanjang;
  
  Validator(this.minPanjang);
  
  // Call method dengan multiple parameters
  bool call(String teks, {bool harusHurufBesar = false}) {
    if (teks.length < minPanjang) return false;
    if (harusHurufBesar && teks != teks.toUpperCase()) return false;
    return true;
  }
}

class Calculator {
  // Overloading call method (berbeda parameter)
  int call(int a, int b) => a + b;
  
  double call.withDouble(double a, double b) => a + b;
  
  int call.kali(int a, int b) => a * b;
}

class Logger {
  final String prefix;
  
  Logger(this.prefix);
  
  void call(String pesan, {String level = 'INFO'}) {
    print('[$level] $prefix: $pesan');
  }
}

void main() {
  print('--- GREETER ---');
  var salamPagi = Greeter('Selamat pagi');
  var salamSore = Greeter('Selamat sore');
  
  // Memanggil objek seperti fungsi
  print(salamPagi('Budi'));    // Output: Selamat pagi, Budi!
  print(salamSore('Ani'));     // Output: Selamat sore, Ani!
  
  print('\n--- VALIDATOR ---');
  var validator = Validator(5);
  print('test: ${validator("test")}');  // false (kurang dari 5)
  print('testing: ${validator("testing")}');  // true
  print('TESTING: ${validator("TESTING", harusHurufBesar: true)}');  // true
  
  print('\n--- CALCULATOR ---');
  var calc = Calculator();
  print('5 + 3 = ${calc(5, 3)}');  // Memanggil objek
  
  print('\n--- LOGGER ---');
  var log = Logger('App');
  log('Aplikasi dimulai');
  log('Database error', level: 'ERROR');
}

8️⃣ GENERIC CLASS DENGAN ENKAPSULASI LANJUTAN

Pengertian: Kelas yang bisa bekerja dengan berbagai tipe data (type-safe) dengan tetap menerapkan enkapsulasi.

Kegunaan: Membuat struktur data yang fleksibel namun tetap aman (type-safe).

💡 Manfaat: Kode lebih reusable, type-safe, dan menghindari runtime error.
📁 generic_class.dart DART 
/// Repository generik dengan enkapsulasi
class Repository<T> {
  final List<T> _items = [];
  final Map<String, T> _cache = {};
  
  // Getter untuk read-only view
  List<T> get semua => List.unmodifiable(_items);
  
  int get jumlah => _items.length;
  
  void tambah(T item) {
    _items.add(item);
    _validateItem(item);
  }
  
  void _validateItem(T item) {
    print('✅ Item type ${T.toString()} ditambahkan');
  }
  
  T? cari(bool Function(T) predicate) {
    try {
      return _items.firstWhere(predicate);
    } catch (e) {
      return null;
    }
  }
  
  void cache(String key, T value) {
    _cache[key] = value;
  }
  
  T? getCache(String key) => _cache[key];
}

/// Pair class dengan dua tipe generik
class Pair<A, B> {
  final A first;
  final B second;
  
  const Pair(this.first, this.second);
  
  Pair<A, B> copyWith({A? first, B? second}) {
    return Pair<A, B>(
      first ?? this.first,
      second ?? this.second,
    );
  }
  
  @override
  String toString() => '($first, $second)';
}

/// Result class untuk operation results
class Result<T> {
  final bool isSuccess;
  final T? data;
  final String? error;
  
  Result._({required this.isSuccess, this.data, this.error});
  
  factory Result.success(T data) {
    return Result._(isSuccess: true, data: data);
  }
  
  factory Result.failure(String error) {
    return Result._(isSuccess: false, error: error);
  }
}

void main() {
  print('--- REPOSITORY GENERIK ---');
  var repoString = Repository<String>();
  repoString.tambah('Data 1');
  repoString.tambah('Data 2');
  
  var repoAngka = Repository<int>();
  repoAngka.tambah(100);
  repoAngka.tambah(200);
  
  print('String items: ${repoString.semua}');
  print('Angka items: ${repoAngka.semua}');
  
  print('\n--- PAIR GENERIK ---');
  var pair1 = Pair<String, int>('Usia', 25);
  var pair2 = Pair<int, String>(1, 'Satu');
  
  print(pair1);
  print(pair2);
  
  print('\n--- RESULT GENERIK ---');
  Result<String> result1 = Result.success('Data berhasil');
  Result<int> result2 = Result.failure('Gagal memuat data');
  
  if (result1.isSuccess) {
    print('Sukses: ${result1.data}');
  }
}

9️⃣ EXTENSION METHODS PADA CLASS LANJUTAN

Pengertian: Menambahkan fungsionalitas baru ke kelas yang sudah ada (termasuk kelas built-in) tanpa mengubah kode aslinya.

Kegunaan: Memperkaya kelas tanpa inheritance, menambah method utility.

💡 Kelebihan: Bisa menambah method ke String, int, List, atau kelas apapun tanpa mengubah kode asli!
📁 extension_methods.dart DART 
// Extension pada String
extension StringExtension on String {
  String get kapitalisasi {
    if (isEmpty) return this;
    return this[0].toUpperCase() + substring(1).toLowerCase();
  }
  
  String get balik => split('').reversed.join('');
  
  int get jumlahVokal {
    final vokal = 'aiueoAIUEO';
    return split('').where((c) => vokal.contains(c)).length;
  }
  
  bool get isPalindrom => this == balik;
  
  String censored({int start = 3, int end = 3}) {
    if (length <= start + end) return '*' * length;
    return substring(0, start) + 
           '*' * (length - start - end) + 
           substring(length - end);
  }
}

// Extension pada int
extension IntExtension on int {
  bool get isGenap => this % 2 == 0;
  bool get isGanjil => !isGenap;
  
  bool get isPrima {
    if (this < 2) return false;
    for (int i = 2; i <= this / 2; i++) {
      if (this % i == 0) return false;
    }
    return true;
  }
  
  List<int> get faktor {
    return [for (int i = 1; i <= this; i++) if (this % i == 0) i];
  }
  
  String toRupiah() => 'Rp ${toString().replaceAllMapped(
    RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'), (m) => '${m[1]}.'
  )}';
}

// Extension pada List
extension ListExtension<T> on List<T> {
  T? get firstOrNull => isEmpty ? null : first;
  T? get lastOrNull => isEmpty ? null : last;
  
  List<T> shuffleList() {
    var list = [...this];
    list.shuffle();
    return list;
  }
  
  List<T> ambilAcak(int jumlah) {
    if (jumlah >= length) return [...this];
    var result = List<T>.empty(growable: true);
    var temp = [...this];
    temp.shuffle();
    for (int i = 0; i < jumlah; i++) {
      result.add(temp[i]);
    }
    return result;
  }
}

void main() {
  print('--- EXTENSION STRING ---');
  String teks = 'dart programming';
  print('Kapitalisasi: ${teks.kapitalisasi}');
  print('Balik: ${teks.balik}');
  print('Jumlah vokal: ${teks.jumlahVokal}');
  
  String email = 'user@email.com';
  print('Email di-censor: ${email.censored(start: 2, end: 8)}');
  
  print('\n--- EXTENSION INT ---');
  int angka = 17;
  print('17 genap? ${angka.isGenap}');
  print('17 prima? ${angka.isPrima}');
  print('Faktor 12: ${12.faktor}');
  print('1000000: ${1000000.toRupiah()}');
  
  print('\n--- EXTENSION LIST ---');
  List<String> buah = ['Apel', 'Mangga', 'Jeruk', 'Pisang', 'Anggur'];
  print('First: ${buah.firstOrNull}');
  print('Shuffled: ${buah.shuffleList()}');
  print('Ambil 2 acak: ${buah.ambilAcak(2)}');
}
