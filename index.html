class Mobil {
  String? merk;
  int? tahun;

  // Parameterized Constructor
  Mobil(this.merk, this.tahun);

  // Named Constructor: Mobil Baru
  Mobil.baru() {
    merk = 'Nissan Skyline GT-R R34';
    tahun = 2002;
  }

  // Named Constructor: dari JSON
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  void infoMobil() {
    print('   Merk : $merk');
    print('   Tahun: $tahun');
    print('   --------------------');
  }
}

  mobil1.infoMobil();

  // 2. Named Constructor baru()
  print('[2] Named Constructor (baru()):');
  Mobil mobil2 = Mobil.baru();
  mobil2.infoMobil();

  // 3. Named Constructor dariJson()
  print('[3] Named Constructor (dariJson()):');
  Map<String, dynamic> dataMobil = {'merk': 'Honda', 'tahun': 2020};

📐 Kelas Titik - Initializer List

Merepresentasikan koordinat dengan atribut x dan y, menggunakan initializer list untuk memproses nilai (menjadikan absolut) sebelum objek dibuat.

titik.dart

Dart 
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : SI2A

class Titik {
  final double x, y;

  // Initializer list untuk memastikan koordinat bernilai absolut
  Titik(double x, double y)
      : x = x.abs(),
        y = y.abs() {
    print('   ✅ Titik (${this.x}, ${this.y}) berhasil dibuat.');
  }

  // Method untuk menghitung jarak dari titik (0,0)
  double jarakDariNol() {
    return (x * x + y * y).sqrt();
  }
}

// Extension untuk menghitung akar kuadrat
extension Sqrt on double {
  double sqrt() => this > 0 ? this * this : 0;
}

void main() {
  print('==================================================');
  print('PROGRAM 4: DEMO KELAS TITIK');
  print('==================================================');

  print('\n[1] Membuat titik dengan koordinat positif:');
  Titik titik1 = Titik(5.0, 7.0);
  print('   Titik1: (${titik1.x}, ${titik1.y})');
  print('   Jarak dari (0,0): ${titik1.jarakDariNol().toStringAsFixed(2)}');

  print('\n[2] Membuat titik dengan koordinat negatif:');
  Titik titik2 = Titik(-3.5, -8.0);
  print('   Titik2: (${titik2.x}, ${titik2.y})');
  print('   Jarak dari (0,0): ${titik2.jarakDariNol().toStringAsFixed(2)}');

  print('\nKesimpulan:');
  print('- Initializer list dieksekusi SEBELUM constructor body');
  print('- Berguna untuk validasi atau manipulasi nilai awal');
  print('- Field final dapat diinisialisasi di initializer list');
  print('==================================================');
}

  String nim;

👤 Kelas Person - Enkapsulasi Dasar

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006

class Person {
  String _nama; // Private field
  int _umur;    // Private field

  Person(this._nama, this._umur);

  // Getter
  String get nama => _nama;
  int get umur => _umur;

  // Setter dengan validasi
  set nama(String namaBaru) {
    if (namaBaru.isNotEmpty) {
      _nama = namaBaru;
      print('   ✅ Nama berhasil diubah menjadi "$namaBaru"');
    } else {
      print('   ❌ Error: Nama tidak boleh kosong!');
    }
  }

  set umur(int umurBaru) {
    if (umurBaru > 0 && umurBaru < 150) {
      _umur = umurBaru;
      print('   ✅ Umur berhasil diubah menjadi $umurBaru tahun');
    } else {
      print('   ❌ Error: Umur harus antara 1-149 tahun! (Input: $umurBaru)');
    }
  }

  void info() {
    print('   Informasi Person: $_nama ($_umur tahun)');
  }
}

void main() {
  print('==================================================');
  print('PROGRAM 6: DEMO KELAS PERSON (ENKAPSULASI)');
  print('==================================================');

  // Membuat objek Person
  print('\n[1] Membuat objek Person:');
  Person person1 = Person('Citra Dewi', 28);
  person1.info();

  // Demonstrasi Getter dan Setter
  print('\n[2] Demonstrasi Getter & Setter:');
  print('   Nama awal: ${person1.nama}');
  print('   Umur awal: ${person1.umur}');
  
  person1.nama = 'Citra Lestari';
  person1.umur = 29;
  print('   Setelah diubah: ${person1.nama} (${person1.umur} tahun)');

  print('\n[3] Demonstrasi Validasi:');
  person1.nama = ''; // Gagal
  person1.umur = 200; // Gagal

  print('\nKesimpulan:');
  print('- Field private tidak bisa diakses langsung dari luar kelas');

🔐 Kelas User - Sistem Login

Merepresentasikan akun dengan atribut username, password, dan email. Menerapkan enkapsulasi dengan metode login dan ganti password yang memiliki validasi berlapis.

user.dart

Dart 
// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : SI2A

class User {
  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email);

  // Getter (password tidak disediakan getter)
  String get username => _username;
  String get email => _email;

  // Method Login
  bool login(String username, String password) {
    print('\n   🔐 Percobaan Login: "$username"');
    if (username == _username && password == _password) {
      print('   ✅ Login BERHASIL! Selamat datang $_username.');
      return true;
    } else {
      print('   ❌ Login GAGAL! Username atau password salah.');
      return false;
    }
  }

  // Method Ganti Password dengan validasi
  void gantiPassword(String passwordLama, String passwordBaru) {
    print('\n   🔑 Percobaan Ganti Password:');
    
    // Validasi password lama
    if (passwordLama != _password) {
      print('   ❌ Gagal: Password lama salah!');
      return;
    }
    
    // Validasi password baru
    if (passwordBaru.length < 6) {
      print('   ❌ Gagal: Password baru minimal 6 karakter!');
      return;
    }
    
    if (passwordBaru == _password) {
      print('   ❌ Gagal: Password baru harus berbeda!');
      return;
    }
    
    _password = passwordBaru;
    print('   ✅ Password BERHASIL diubah!');
  }

  void infoUser() {
    print('   👤 Username: $_username');
    print('   📧 Email   : $_email');
    print('   🔒 Password: [PROTECTED]');
  }
}

void main() {
  print('==================================================');
  print('PROGRAM 8: DEMO KELAS USER (LOGIN SYSTEM)');
  print('==================================================');

  // Membuat objek User
  print('\n[1] Membuat user baru:');
  User user1 = User('dini_putri', 'rahasia123', 'dini@email.com');
  user1.infoUser();

  // Demonstrasi Login
  print('\n[2] DEMONSTRASI LOGIN:');
  user1.login('dini_putri', 'rahasia123'); // Berhasil
  user1.login('dini_putri', 'salah');      // Gagal

  // Demonstrasi Ganti Password
  print('\n[3] DEMONSTRASI GANTI PASSWORD:');
  user1.gantiPassword('rahasia123', 'rahasia456'); // Berhasil
  user1.login('dini_putri', 'rahasia456');         // Login dengan password baru

  print('\nKesimpulan:');
  print('- Password disembunyikan (tidak ada getter untuk password)');
  print('- Method login memvalidasi kredensial sebelum akses');
  print('- Ganti password memiliki validasi berlapis');
  print('==================================================');
}
