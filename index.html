🚗 Kelas Mobil - Named Constructor

Merepresentasikan mobil dengan parameterized constructor dan named constructor.

mobil.dart

// Nama   : USMAN YAKIN TRIO SINADA
// Nim    : 251410006
// Kelas  : SI2A

class Mobil {
  String? merk;
  int? tahun;
  
  Mobil(this.merk, this.tahun);
  Mobil.baru() { merk = 'Nissan Skyline'; tahun = 2002; }
  
  void infoMobil() => print('Merk: $merk, Tahun: $tahun');
}

void main() {
  Mobil mobil1 = Mobil('Toyota', 2018);
  mobil1.infoMobil();
  
  Mobil mobil2 = Mobil.baru();
  mobil2.infoMobil();
}
